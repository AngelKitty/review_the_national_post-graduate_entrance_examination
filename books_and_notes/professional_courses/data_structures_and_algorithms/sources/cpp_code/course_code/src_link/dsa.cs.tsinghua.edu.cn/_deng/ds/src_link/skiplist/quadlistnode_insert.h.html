0009 template <typename T> QlistNodePosi(T) //将e作为当前节点的后继、b的上邻插入Quadlist
0010 QuadlistNode<T>::insertAsSuccAbove ( T const& e, QlistNodePosi(T) b = NULL ) {
0011    QlistNodePosi(T) x = new QuadlistNode<T> ( e, this, succ, NULL, b ); //创建新节点
0012    succ->pred = x; succ = x; //设置水平逆向链接
0013    if ( b ) b->above = x; //设置垂直逆向链接
0014    return x; //返回新节点的位置
0015 }
