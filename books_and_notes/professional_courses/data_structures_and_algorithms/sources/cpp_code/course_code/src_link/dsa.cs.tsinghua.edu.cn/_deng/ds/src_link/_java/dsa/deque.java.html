0010  * 双端队列接口
0011  */
0012 
0013 package dsa;
0014 
0015 public interface Deque {
0016    public int getSize();//返回队列中元素数目
0017    public boolean isEmpty();//判断队列是否为空
0018    public Object first() throws ExceptionQueueEmpty;//取首元素（但不删除）
0019    public Object last() throws ExceptionQueueEmpty;//取末元素（但不删除）
0020    public void insertFirst(Object obj);//将新元素作为首元素插入
0021    public void insertLast(Object obj);//将新元素作为末元素插入
0022    public Object removeFirst() throws ExceptionQueueEmpty;//删除首元素
0023    public Object removeLast() throws ExceptionQueueEmpty;//删除末元素
0024    public void Traversal();//遍历
0025 }
