0009 void shuffle ( int A[], int n ) { //将A[0, n)随机打乱
0010    while ( 1 < n )
0011       swap ( A[rand() % n], A[--n] );
0012 }
0013 
0014 void shuffle ( int A[], int lo, int hi ) //将A[lo, hi)随机打乱
0015 {  shuffle ( A + lo, hi - lo );  }
