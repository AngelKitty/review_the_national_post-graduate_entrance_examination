0001 #include <stdlib.h>
0002 
0003 extern int s_lo, s_hi;
0004 
0005 int gs_DC( int A[], int lo, int hi ) { //分治策略：O(n*logn)
0006    if ( hi - lo < 2 ) return A[lo];
0007    int mi = (lo + hi) / 2;
0008 
0009    int gsL = A[mi-1], sL = 0, i = mi; //枚举
0010    while ( lo < i-- ) //所有[i, mi)类区段
0011       if ( gsL < (sL += A[i]) ) gsL = sL;
0012 
0013    int gsR = A[mi], sR = 0, j = mi-1; //枚举
0014    while ( ++j < hi ) //所有[mi, j)类区段
0015       if ( gsR < (sR += A[j]) ) gsR = sR; //择优、更新
0016 
0017    return __max( gsL + gsR, __max( gs_DC( A, lo, mi ), gs_DC( A, mi, hi ) ) ); //递归
0018 }
