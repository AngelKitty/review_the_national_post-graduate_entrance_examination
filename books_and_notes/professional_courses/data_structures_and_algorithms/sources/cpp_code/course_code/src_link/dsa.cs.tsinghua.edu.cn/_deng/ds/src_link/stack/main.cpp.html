0012 #include "stack_test.h"
0013 
0014 int testID = 0; //测试编号
0015 
0016 /******************************************************************************************
0017  * 测试栈
0018  ******************************************************************************************/
0019 template <typename T> //元素类型
0020 void testStack ( int n ) {
0021    Stack<T> S;
0022    printf ( "\n  ==== Test %2d. Growing stack\n", testID++ );
0023    while ( S.size() < n ) {
0024       if ( S.empty() || ( 30 < ( rand() % 100 ) ) ) { //70%概率入栈
0025          T e = (T) dice ( 2 * n );
0026          printf ( "pushing " ); print ( e ); printf ( " ...\n" );
0027          S.push ( e );
0028       } else { //30%概率出栈
0029          printf ( "popping with ... " ); print ( S.pop() ); printf ( "\n" );
0030       }
0031       print ( S );
0032    }
0033    printf ( "\n  ==== Test %2d. Shrinking stack\n", testID++ );
0034    while ( !S.empty() ) {
0035       if ( 70 < dice ( 100 ) ) { //70%概率入栈
0036          T e = (T) dice ( 2 * n );
0037          printf ( "pushing " ); print ( e ); printf ( " ...\n" );
0038          S.push ( e );
0039       } else { //70%概率出栈
0040          printf ( "popping with ... " ); print ( S.pop() ); printf ( "\n" );
0041       }
0042       print ( S );
0043    }
0044 }
0045

0047 /******************************************************************************************
0048  * 测试栈
0049  ******************************************************************************************/
0050 int main ( int argc, char* argv[] ) {
0051    if ( 2 > argc ) { printf ( "Usage: %s <size of test>\a\a\n", argv[0] ); return 1; }
0052    srand ( ( unsigned int ) time ( NULL ) ); //设置随机种子
0053    testStack<int> ( atoi ( argv[1] ) ); //元素类型可以在这里任意选择
0054    return 0;
0055 }
