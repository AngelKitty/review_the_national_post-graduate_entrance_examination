0011 /******************************************************************************************
0012  * 显示表达式片段exp[lo, hi]，与原表达式对齐
0013  ******************************************************************************************/
0014 void displaySubstring ( const char exp[], int lo, int hi ) {
0015    for ( int i = 0; i < lo; i++ ) printf ( " " );
0016    for ( int i = lo; i <= hi; i++ ) printf ( "%c", exp[i] );
0017    printf ( "\n" );
0018 }

0020 /******************************************************************************************
0021  * 显示表达式扫描进度
0022  ******************************************************************************************/
0023 void displayProgress ( const char exp[], int i, Stack<char> S ) {

0027    for ( int j = 0; j < i; j++ )   printf ( "-" );
