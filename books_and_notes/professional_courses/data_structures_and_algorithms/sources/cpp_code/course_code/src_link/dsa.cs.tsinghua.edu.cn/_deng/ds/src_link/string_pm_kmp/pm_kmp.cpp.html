0009 int match ( char* P, char* T ) {  //KMP算法
0010    int* next = buildNext ( P ); //构造next表
0011    int n = ( int ) strlen ( T ), i = 0; //文本串指针
0012    int m = ( int ) strlen ( P ), j = 0; //模式串指针
0013    while ( j < m  && i < n ) //自左向右逐个比对字符
0014       if ( 0 > j || T[i] == P[j] ) //若匹配，或P已移出最左侧（两个判断的次序不可交换）
0015          { i ++;  j ++; } //则转到下一字符
0016       else //否则
0017          j = next[j]; //模式串右移（注意：文本串不用回退）
0018    delete [] next; //释放next表
0019    return i - j;
0020 }
