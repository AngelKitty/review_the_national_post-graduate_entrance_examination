0009 int match ( char* P, char* T ) { //Boyer-Morre算法（完全版，兼顾Bad Character与Good Suffix）
0010    int* bc = buildBC ( P ); int* gs = buildGS ( P ); //构造BC表和GS表
0011    size_t i = 0; //模式串相对于文本串的起始位置（初始时与文本串左对齐）
0012    while ( strlen ( T ) >= i + strlen ( P ) ) { //不断右移（距离可能不止一个字符）模式串
0013       int j = strlen ( P ) - 1; //从模式串最末尾的字符开始
0014       while ( P[j] == T[i + j] ) //自右向左比对
0015          if ( 0 > --j ) break;
0016       if ( 0 > j ) //若极大匹配后缀 == 整个模式串（说明已经完全匹配）
0017          break; //返回匹配位置
0018       else //否则，适当地移动模式串
0019          i += __max ( gs[j], j - bc[ T[i + j] ] ); //位移量根据BC表和GS表选择大者
0020    }
0021    delete [] gs; delete [] bc; //销毁GS表和BC表
0022    return i;
0023 }
