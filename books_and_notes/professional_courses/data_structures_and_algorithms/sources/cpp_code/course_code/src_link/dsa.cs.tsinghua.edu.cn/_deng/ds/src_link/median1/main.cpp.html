0020 /******************************************************************************************
0021  * 中位数算法测试
0022  ******************************************************************************************/
0023 int main ( int argc, char* argv[] ) {
0024 // 解析命令行
0025    if ( 7 > argc ) { printf ( "Usage:\n\t%s <#tests> <n> <min1> <max1> <min2> <max2>\a\a\n", argv[0] ); return 1; }
0026    int nTest = atoi ( argv[1] ); //测试轮数
0027    int n = atoi ( argv[2] ); //向量长度
0028    int min1 = atoi ( argv[3] ), max1 = atoi ( argv[4] ); //数值范围
0029    int min2 = atoi ( argv[5] ), max2 = atoi ( argv[6] ); //数值范围
0030    if ( ( nTest < 1 ) || ( n < 1 ) || ( max1 + 1 < min1 + n ) || ( max2 + 1 < min2 + n ) )
0031       { printf ( "arguments error\a\a\n" ); return 1; }
0032 // 随机种子
0033    srand ( ( unsigned int ) time ( NULL ) );
0034 // 每次测试100组
0035    for ( int i = 0; i < nTest; i++ ) {
0036       // 随机生成向量
0037       Vector<int> A1 = randomSortedVector ( n, min1, max1 ); print ( A1 );
0038       Vector<int> A2 = randomSortedVector ( n, min2, max2 ); print ( A2 );
0039       // 计算中位数，并与蛮力算法（更不易出错）的结果对比
0040       int m1 = median ( A1, 0, A2, 0, n );
0041       int m2 = trivialMedian ( A1, 0, n, A2, 0, n );
0042       if ( m1 == m2 ) { printf ( "\nmedian: %d = %d\a\n\n\n", m1, m2 ); }
0043       else          { printf ( "\nmedian: %d <Error> %d\a\a\n\n\n", m1, m2 ); getchar(); }
0044    }
0045    // 返回
0046    return 0;
0047 }
