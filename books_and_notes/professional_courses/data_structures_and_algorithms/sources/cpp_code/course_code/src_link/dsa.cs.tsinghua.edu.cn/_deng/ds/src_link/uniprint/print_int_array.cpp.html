0009 #include <stdio.h>
0010 
0011 void print ( int A[], int n, int lo, int hi ) { //输出整数数组区间A[lo, hi)
0012    bool sorted = true;
0013    for ( int i = 1; i < n; i++ ) if ( A[i-1] > A[i] ) sorted = false;
0014    printf ( sorted ? "S: " : " : " );
0015    for ( int i = 0; i < lo; i++ ) printf ( "   ." );
0016    for ( int i = lo; i < hi; i++ ) printf ( "%4d", A[i] );
0017    for ( int i = hi; i < n; i++ ) printf ( "   ." );
0018    printf ( "\n" );
0019 }
0020 
0021 void print ( int A[], int n ) { print ( A, n, 0, n ); }
