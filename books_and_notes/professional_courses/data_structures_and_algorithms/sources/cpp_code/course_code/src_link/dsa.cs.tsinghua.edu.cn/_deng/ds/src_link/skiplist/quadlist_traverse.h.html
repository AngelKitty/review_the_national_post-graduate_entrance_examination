0009 template <typename T> //遍历Quadlist，对各节点依次实施visit操作
0010 void Quadlist<T>::traverse ( void ( *visit ) ( T& ) ) { //利用函数指针机制，只读或局部性修改
0011    QlistNodePosi(T) p = header;
0012    while ( ( p = p->succ ) != trailer ) visit ( p->data );
0013 }
