0015 int match ( char* P, char* T ) { //串匹配算法（Brute-force-1）
0016    size_t n = strlen ( T ), i = 0; //文本串长度、当前接受比对字符的位置
0017    size_t m = strlen ( P ), j = 0; //模式串长度、当前接受比对字符的位置
0018    while ( j < m && i < n ) //自左向右逐个比对字符
0019       if ( T[i] == P[j] ) //若匹配
0020          { i ++;  j ++; } //则转到下一对字符
0021       else //否则
0022          { i -= j - 1; j = 0; } //文本串回退、模式串复位
0023    return i - j; //如何通过返回值，判断匹配结果？
0024 }
