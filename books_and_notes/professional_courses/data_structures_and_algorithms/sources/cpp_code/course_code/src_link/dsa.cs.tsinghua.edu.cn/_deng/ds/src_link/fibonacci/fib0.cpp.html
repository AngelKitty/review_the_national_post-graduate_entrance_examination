0009 __int64 fibI ( int n ) { //计算Fibonacci数列的第n项（迭代版）：O(n)
0010    __int64 f = 1, g = 0; //初始化：fib(-1)、fib(0)
0011    while ( 0 < n-- ) { g += f; f = g - f; } //依据原始定义，通过n次加法和减法计算fib(n)
0012    return g; //返回
0013 }
