0010  * 向量、列表等支持traverse()遍历操作的线性结构
0011  ******************************************************************************************/
0012 template <typename T> //元素类型
0013 void UniPrint::p ( T& s ) { //引用
0014    printf ( "%s[%d]*%d:\n", typeid ( s ).name(), &s, s.size() ); //基本信息
0015    s.traverse ( print ); //通过print()遍历输出所有元素
0016    printf ( "\n" );
0017 }
