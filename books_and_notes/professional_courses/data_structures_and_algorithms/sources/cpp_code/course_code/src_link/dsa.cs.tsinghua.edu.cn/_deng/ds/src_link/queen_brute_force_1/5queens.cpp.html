0011 void place5Queens_BruteForce() { //5皇后蛮力算法
0012    int solu[5]; //解法
0013    int nCandidate = 0; //候选解总数
0014    for ( solu[0] = 0; solu[0] < 5; solu[0]++ )
0015       for ( solu[1] = 0; solu[1] < 5; solu[1]++ )
0016          for ( solu[2] = 0; solu[2] < 5; solu[2]++ )
0017             for ( solu[3] = 0; solu[3] < 5; solu[3]++ )
0018                for ( solu[4] = 0; solu[4] < 5; solu[4]++ ) {
0019                   nCandidate++;
0020                   if ( collide ( solu, 0 ) ) continue;
0021                   if ( collide ( solu, 1 ) ) continue;
0022                   if ( collide ( solu, 2 ) ) continue;
0023                   if ( collide ( solu, 3 ) ) continue;
0024                   if ( collide ( solu, 4 ) ) continue;
0025                   nSolu++; displaySolution ( solu, 5 );
0026                }
0027    printf ( "%d Queens, %d Solution(s), %d Candidate(s), %dCheck(s)\n\a\a", 5, nSolu, nCandidate, nCheck ); //输出解的总数
0028 }
