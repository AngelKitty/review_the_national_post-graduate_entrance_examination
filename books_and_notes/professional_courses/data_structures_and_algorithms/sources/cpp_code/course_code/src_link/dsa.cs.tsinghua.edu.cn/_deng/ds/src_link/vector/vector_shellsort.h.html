0010 void Vector<T>::shellSort ( Rank lo, Rank hi ) { //0 <= lo < hi <= size <= 2^30
0011    for ( int d = 0x3FFFFFFF; 0 < d; d >>= 1 ) //PS Sequence: { 1, 3, 7, 15, ..., 1073741823 }
0012       for ( int j = lo + d; j < hi; j++ ) { //for each j in [lo+d, hi)
0013          T x = _elem[j]; int i = j - d;
0014          while ( lo <= i && _elem[i] > x )
0015             { _elem[i + d] = _elem[i]; i -= d; }
0016          _elem[i + d] = x; //insert [j] into its subsequence
0017       }
0018 }
