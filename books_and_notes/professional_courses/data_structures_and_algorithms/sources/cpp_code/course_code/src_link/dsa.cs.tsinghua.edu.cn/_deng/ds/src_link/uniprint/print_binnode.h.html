0009 /******************************************************************************************
0010  * BinTree节点
0011  ******************************************************************************************/
0012 template <typename T> void UniPrint::p ( BinNode<T>& node ) {
0013    p ( node.data ); //数值
0014    /******************************************************************************************
0015     * height & NPL
0016     ******************************************************************************************/
0017 #if   defined(DSA_LEFTHEAP)
0018    printf ( "(%-2d)", node.npl ); //NPL
0019 #elif defined(DSA_BST)
0020    printf ( "(%-2d)", node.height ); //高度
0021 #elif defined(DSA_AVL)
0022    printf ( "(%-2d)", node.height ); //高度
0023 #elif defined(DSA_REDBLACK)
0024    printf ( "(%-2d)", node.height ); //高度
0025 #elif defined(DSA_SPLAY)
0026    printf ( "(%-2d)", node.height ); //高度
0027 #endif
0028    /******************************************************************************************
0029     * 父子链接指针
0030     ******************************************************************************************/
0031    printf (
0032       (  ( node.lc && &node != node.lc->parent ) ||
0033          ( node.rc && &node != node.rc->parent )  ) ?
0034       "@" : " "
0035    );
0036    /******************************************************************************************
0037     * 节点颜色
0038     ******************************************************************************************/
0039 #if defined(DSA_REDBLACK)
0040    printf ( node.color == RB_BLACK ? "B" : " " ); //（忽略红节点）
0041 #endif
0042    /******************************************************************************************
0043     * 父子（黑）高度、NPL匹配
0044     ******************************************************************************************/
0045 #if   defined(DSA_PQ_COMPLHEAP)
0046    //高度不必匹配
0047 #elif defined(DSA_PQ_LEFTHEAP)
0048    printf ( //NPL
0049       ( node.rc && node.npl != 1 + node.rc->npl ) ||
0050       ( node.lc && node.npl >  1 + node.lc->npl ) ?
0051       "%%" : " "
0052    );
0053 #elif defined(DSA_REDBLACK)
0054    printf ( BlackHeightUpdated ( node ) ? " " : "!" ); //黑高度
0055 #else
0056    printf ( HeightUpdated ( node ) ? " " : "!" ); //（常规）高度
0057 #endif
0058    /******************************************************************************************
0059     * 左右平衡
0060     ******************************************************************************************/
0061 #if defined(DSA_AVL)
0062    if ( !AvlBalanced ( node ) ) printf ( "X" ); //AVL平衡
0063    else if ( 0 < BalFac ( node ) ) printf ( "\&#92;&#34; ); //AVL平衡
0064    else if ( BalFac ( node ) < 0 ) printf ( "/" ); //AVL平衡
0065    else printf ( "-" ); //AVL平衡
0066 #elif defined(DSA_REDBLACK)
0067    if ( !Balanced ( node ) ) printf ( "X" ); //RB平衡
0068    else if ( 0 < BalFac ( node ) ) printf ( "\&#92;&#34; ); //RB平衡
0069    else if ( BalFac ( node ) < 0 ) printf ( "/" ); //RB平衡
0070    else printf ( "-" ); //RB平衡
0071 #else
0072    //平衡无所谓
0073 #endif
0074 }
