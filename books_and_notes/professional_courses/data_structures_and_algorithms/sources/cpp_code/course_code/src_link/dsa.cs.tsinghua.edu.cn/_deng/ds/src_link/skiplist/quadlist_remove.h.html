0009 template <typename T> //删除Quadlist内位置p处的节点，返回其中存放的词条
0010 T Quadlist<T>::remove ( QlistNodePosi(T) p ) { //assert: p为Quadlist中的合法位置
0011    p->pred->succ = p->succ; p->succ->pred = p->pred; _size--;//摘除节点
0012    T e = p->entry; delete p; //备份词条，释放节点
0013    return e; //返回词条
0014 }
