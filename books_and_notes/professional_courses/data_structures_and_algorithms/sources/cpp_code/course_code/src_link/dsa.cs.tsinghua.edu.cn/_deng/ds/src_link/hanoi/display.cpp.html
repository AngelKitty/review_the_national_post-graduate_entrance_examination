0012 /******************************************************************************************
0013  * 输出栈S在高度i的分布
0014  ******************************************************************************************/
0015 void display ( Stack<int>& S, int i ) {
0016    int diskR = ( i < S.size() ) ? S[i] : 0; //通过Vecotr[]接口，获取高度i处盘子的半径
0017    for ( int j = 0; j < nDisk - diskR; j++ )   printf ( " " );
0018    for ( int j = 0; j < diskR; j++ )    printf ( "=" );
0019    printf ( "|" );
0020    for ( int j = 0; j < diskR; j++ )    printf ( "=" );
0021    for ( int j = 0; j < nDisk - diskR; j++ )   printf ( " " );
0022 }
0023 
0024 /******************************************************************************************
0025  * 显示三个栈内容（使用全局变量）
0026  ******************************************************************************************/
0027 void displayHanoi() {
0028    system ( "cls" );   printf ( "\n\n" );
0029    for ( int i = nDisk - 1; i >= 0; i-- ) //自顶向下，在每一高度层次，输出（至多）三个盘子
0030       { display ( Sx, i ); display ( Sy, i ); display ( Sz, i ); printf ( "\n" ); }
0031    for ( int j = 0; j < 3 * ( 2 * nDisk + 1 ); j++ )   printf ( "=" );   printf ( "\n" ); //显示地面
0032    getchar();
0033 }
