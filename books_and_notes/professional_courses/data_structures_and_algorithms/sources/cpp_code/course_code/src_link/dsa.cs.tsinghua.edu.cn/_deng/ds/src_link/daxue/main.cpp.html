0009 #include<iostream>
0010 using namespace std;
0011 
0012 #define nStep  8
0013 char* level[8] =
0014 {"明明德於天下", "治", "R家", "修身", "正心", "\意", "致知", "格物"};
0015 
0016 void phase ( int n, bool done ) {
0017    for ( int i = 0; i < n; i++ )  cout << "\t│";
0018    cout << "\t" << ( done ? "└" : "┌" ) << level[n] << endl;
0019 }
0020 
0021 void towardsDaxue ( int n ) {
0022    phase ( n, false );
0023    if ( n + 1 < nStep ) towardsDaxue ( n + 1 );
0024    phase ( n, true );
0025 }
0026 
0027 int main ( char* argv[], int argc ) {
0028    towardsDaxue ( 0 );
0029    return 0;
