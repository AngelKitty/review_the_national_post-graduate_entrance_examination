0013 package dsa;
0014 
0015 public class Sorter_Bubblesort implements Sorter {
0016    private Comparator C;
0017 
0018    public Sorter_Bubblesort()
0019    {  this(new ComparatorDefault()); }
0020 
0021    public Sorter_Bubblesort(Comparator comp)
0022    {  C = comp; }
0023 
0024    public void sort(Sequence S) {
0025       int n = S.getSize();
0026       for (int i = 0; i < n; i++)
0027          for (int j = 0; j < n - i - 1; j++)
0028             if (0 < C.compare(S.getAtRank(j), S.getAtRank(j + 1))) {
0029                Object temp = S.getAtRank(j);
0030                S.replaceAtRank(j, S.getAtRank(j + 1));
0031                S.replaceAtRank(j + 1, temp);
0032             }
0033    }
0034 }
