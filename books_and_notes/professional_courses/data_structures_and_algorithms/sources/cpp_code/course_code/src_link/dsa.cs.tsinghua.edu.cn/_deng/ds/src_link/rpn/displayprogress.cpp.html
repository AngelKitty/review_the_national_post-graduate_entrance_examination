0018 /******************************************************************************************
0019  * 显示表达式处理进展
0020  ******************************************************************************************/
0021 void displayProgress ( char* expr, char* pCh, Stack<float>& opndStk, Stack<char>& optrStk, char* rpn ) {
0022    system ( "cls" );
0023    printf ( "\n" );
0024    for ( char* p = expr; '\0' != *p; p++ ) printf ( " %c", *p ); printf ( " $\n" );
0025    for ( char* p = expr; p < pCh; p++ ) printf ( "  " );
0026    if ( '\0' != * ( pCh - 1 ) )
0027       { for ( char* p = pCh; '\0' != *p; p++ ) printf ( " %c", *p ); printf ( " $" ); }
0028    printf ( "\n" );
0029    for ( char* p = expr; p < pCh; p++ ) printf ( "--" ); printf ( " ^\n\n" );
0030    print ( optrStk ); printf ( "\n" );
0031    print ( opndStk ); printf ( "\n" );
0032    printf ( "RPN:\n %s\n\n", rpn ); //输出RPN
0033    getchar();
0034 }
