0009 inline Cell* neighbor ( Cell* cell ) { //查询当前位置的相邻格点
0010    switch ( cell->outgoing ) {
0011       case EAST  : return cell + LABY_MAX; //向东
0012       case SOUTH : return cell + 1;        //向南
0013       case WEST  : return cell - LABY_MAX; //向西
0014       case NORTH : return cell - 1;        //向北
0015       default    : exit ( -1 );
0016    }
0017 }
