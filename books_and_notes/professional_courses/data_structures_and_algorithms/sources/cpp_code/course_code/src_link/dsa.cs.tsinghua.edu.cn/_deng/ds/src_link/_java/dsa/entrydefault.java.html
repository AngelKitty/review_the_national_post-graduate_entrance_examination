0009 /*
0010  * 默认条目
0011  */
0012 
0013 package  dsa;
0014 
0015 public class EntryDefault implements Entry {
0016    protected Object key;
0017    protected Object value;
0018 
0019    /**************************** 构造函数 ****************************/
0020    public EntryDefault(Object k, Object v)
0021    { key = k;  value = v; }
0022 
0023    /**************************** Entry接口方法 ****************************/
0024    //取条目的关键码
0025    public Object getKey()
0026    { return key; }
0027    //修改条目的关键码，返回此前存放的关键码
0028    public Object setKey(Object k)
0029    { Object oldK = key; key = k; return oldK; }
0030 
0031    //取条目的数据对象
0032    public Object getValue()
0033    { return value; }
0034    //修改条目的数据对象，返回此前存放的数据对象
0035    public Object setValue(Object v)
0036    { Object oldV = value;  value = v;  return oldV; }
0037 }
