0011 /******************************************************************************************

0013  ******************************************************************************************/
0014 void UniPrint::p ( int e ) { printf ( " %04d", e ); }
0015 void UniPrint::p ( float e ) { printf ( " %4.3f", e ); }
0016 void UniPrint::p ( double e ) { printf ( " %4.3f", e ); }
0017 void UniPrint::p ( char e ) { printf ( " %c", ( 31 < e ) && ( e < 128 ) ? e : '$' ); }
0018 void UniPrint::p ( VStatus e ) {
0019    switch ( e ) {
0020       case UNDISCOVERED:   printf ( "U" ); break;
0021       case DISCOVERED:     printf ( "D" ); break;
0022       case VISITED:        printf ( "V" ); break;
0023       default:             printf ( "X" ); break;
0024    }
0025 }
0026 void UniPrint::p ( EType e ) {
0027    switch ( e ) {
0028       case UNDETERMINED:   printf ( "U" ); break;
0029       case TREE:           printf ( "T" ); break;
0030       case CROSS:          printf ( "C" ); break;
0031       case BACKWARD:       printf ( "B" ); break;
0032       case FORWARD:        printf ( "F" ); break;
0033       default:             printf ( "X" ); break;
0034    }
0035 }
