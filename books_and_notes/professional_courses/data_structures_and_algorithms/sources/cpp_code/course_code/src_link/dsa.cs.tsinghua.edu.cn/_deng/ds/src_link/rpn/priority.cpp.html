0009 Operator optr2rank ( char op ) { //由运算符转译出编号
0010    switch ( op ) {
0011       case '+' : return ADD; //加
0012       case '-' : return SUB; //减
0013       case '*' : return MUL; //乘
0014       case '/' : return DIV; //除
0015       case '^' : return POW; //乘方
0016       case '!' : return FAC; //阶乘
0017       case '(' : return L_P; //左括号
0018       case ')' : return R_P; //右括号
0019       case '\0': return EOE; //起始符与终止符
0020       default  : exit ( -1 ); //未知运算符
0021    }
0022 }
