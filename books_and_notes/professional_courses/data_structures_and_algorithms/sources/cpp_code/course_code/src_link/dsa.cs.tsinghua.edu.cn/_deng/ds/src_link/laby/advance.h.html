0009 inline Cell* advance ( Cell* cell ) { //从当前位置转入相邻格点
0010    Cell* next;
0011    switch ( cell->outgoing ) {
0012       case EAST:  next = cell + LABY_MAX; next->incoming = WEST;  break; //向东
0013       case SOUTH: next = cell + 1;        next->incoming = NORTH; break; //向南
0014       case WEST:  next = cell - LABY_MAX; next->incoming = EAST;  break; //向西
0015       case NORTH: next = cell - 1;        next->incoming = SOUTH; break; //向北
0016       default : exit ( -1 );
0017    }
0018    return next;
0019 }
