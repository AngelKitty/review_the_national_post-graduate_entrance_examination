0018 #define PRIMEFILE "prime-bitmap.txt" //用于记录指定区间内素数分布的比特图文件
0019 int main ( int argc, char* argv[] ) {
0020    srand ( ( unsigned int ) time ( NULL ) ); //设置随机种子
0021    if ( 2 > argc ) { printf ( "Usage: %s <maxInt>\n", argv[0] ); return -1; }
0022    int n = atoi ( argv[1] ); //简化起见，不考虑非正的情况
0023    Eratosthenes ( n, PRIMEFILE );
0024    for ( int i = 0; i < 13; i++ ) { //做13次随机测试（限于MAX_RAND，覆盖范围不超过0x7FFF）
0025       int low = dice ( n );
0026       printf ( "The smallest prime number no less than %8d (%05X) is %8d\n", low, low, primeNLT ( low, n, PRIMEFILE ) );
0027    }
0028    for ( int i = 0; i < 13; i++ ) { //做13次等间距测试
0029       printf ( "The smallest prime number no less than %8d (%05X) is %8d\n", n * i / 13, n * i / 13, primeNLT ( n * i / 13, n, PRIMEFILE ) );
0030    }
0031    return 0;
0032 }
