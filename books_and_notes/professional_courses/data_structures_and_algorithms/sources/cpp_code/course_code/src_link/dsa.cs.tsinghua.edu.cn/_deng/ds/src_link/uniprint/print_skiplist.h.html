0013 template <typename K, typename V> //e、value
0014 void UniPrint::p ( Skiplist<K, V>& s ) { //引用
0015    printf ( "%s[%d]*%d*%d:\n", typeid ( s ).name(), &s, s.level(), s.size() ); //基本信息
0016    s.traverse ( print ); //通过print()遍历输出所有元素
0017    printf ( "\n" );
0018 }
