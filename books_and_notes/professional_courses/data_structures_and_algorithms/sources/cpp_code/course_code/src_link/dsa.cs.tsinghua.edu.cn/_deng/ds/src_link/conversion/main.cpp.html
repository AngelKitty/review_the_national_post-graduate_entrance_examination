0009 #include "convert.h"

0011 /******************************************************************************************
0012  * 进制转换
0013  ******************************************************************************************/
0014 int main ( int argc, char* argv[] ) {
0015    for ( int i = 1; i < argc; i += 2 ) {
0016       __int64 n = _atoi64 ( argv[i] ); //待转换的十进制数
0017       int base = atoi ( argv[i+1] ); //目标进制
0018       Stack<char> S; //用栈记录转换得到的各数位
0019       convert ( S, n, base ); //进制转换
0020       while ( !S.empty() ) printf ( "%c", ( S.pop() ) ); //逆序输出栈内数位，即正确结果
0021    }
0022    return 0;
0023 }
