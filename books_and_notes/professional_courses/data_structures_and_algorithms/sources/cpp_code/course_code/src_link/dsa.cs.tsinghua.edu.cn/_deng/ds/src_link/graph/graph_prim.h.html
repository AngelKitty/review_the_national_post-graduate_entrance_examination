0009 template <typename Tv, typename Te> //Prim算法：无向连通图，各边表示为方向互逆、权重相等的一对边
0010 void Graph<Tv, Te>::prim ( int s ) { //assert: 0 <= s < n
0011    reset(); priority ( s ) = 0;
0012    for ( int i = 0; i < n; i++ ) { //共需引入n个顶点和n-1条边
0013       status ( s ) = VISITED;
0014       if ( -1 != parent ( s ) ) type ( parent ( s ), s ) = TREE; //引入当前的s
0015       for ( int j = firstNbr ( s ); -1 < j; j = nextNbr ( s, j ) ) //枚举s的所有邻居j
0016          if ( ( status ( j ) == UNDISCOVERED ) && ( priority ( j ) > weight ( s, j ) ) ) //对邻接顶点j做松弛
0017             { priority ( j ) = weight ( s, j ); parent ( j ) = s; } //与Dijkstra算法唯一的不同之处
0018       for ( int shortest = INT_MAX, j = 0; j < n; j++ ) //选出下一极短跨边
0019          if ( ( status ( j ) == UNDISCOVERED ) && ( shortest > priority ( j ) ) )
0020             { shortest = priority ( j ); s = j; }
0021    }
0022 }
